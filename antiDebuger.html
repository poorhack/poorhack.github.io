JS 动态 Debugger 拦截脚本

该脚本通过重写 window.Function 构造函数，拦截通过字符串拼接（如 "de" + "bugger"）生成的断点，有效应对复杂的 H5 反调试逻辑。

(function() {
    // 1. 保存原始 Function 的引用（防止死循环）
    const RawFunction = window.Function;

    // 2. 定义统一的拦截逻辑函数
    const hookedFunction = function() {
        // 获取最后一个参数（通常是函数体）
        const body = arguments[arguments.length - 1];
        
        // 检查是否包含 "debugger" 变体
        if (typeof body === 'string' && body.includes('bugger')) {
            console.warn("已拦截动态 debugger 注入");
            return function() {}; // 返回空函数
        }
        
        // 如果是正常调用，使用 RawFunction 构造
        return RawFunction.apply(this, arguments);
    };

    // 3. 修复原型链：确保 ({}).constructor 这种方式也被拦截
    hookedFunction.prototype = RawFunction.prototype;
    hookedFunction.prototype.constructor = hookedFunction;

    // 4. 全局替换
    window.Function = hookedFunction;
    
    console.log("Debugger 拦截系统已整合就绪");
})();
